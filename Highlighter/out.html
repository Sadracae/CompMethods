def factorial(num):
    print("Calculating factorial... ")

    if num < 0:
        return "Factorial is not defined for negative numbers"
    
    result = 1
    
    while num > 1:
        result *= num
        num -= 1 
    
    return result

num = 5
fact = factorial(num)
print("Factorial of", num, "is", fact)
